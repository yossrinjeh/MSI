    // JDBC operations

### 3.2 Database Schema
```sql
CREATE TABLE drawings (
    id INTEGER PRIMARY KEY,
    name VARCHAR(255),
    created_date TIMESTAMP,
    drawing_data TEXT
);

CREATE TABLE shapes (
    id INTEGER PRIMARY KEY,
    drawing_id INTEGER,
    shape_type VARCHAR(50),
    x1 DOUBLE, y1 DOUBLE, x2 DOUBLE, y2 DOUBLE,
    color VARCHAR(20),
    is_filled BOOLEAN,
    FOREIGN KEY (drawing_id) REFERENCES drawings(id)
);
```
